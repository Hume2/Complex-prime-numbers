SOURCES = $(wildcard *.cpp) $(wildcard level/*.cpp) $(wildcard generator/*.cpp) $(wildcard generator/hnus/*.cpp) $(wildcard tema/*.cpp)
OBJECTS = $(SOURCES:.cpp=.o)
DEPS = $(SOURCES:%.cpp=%.d)
TARGET = program

CFLAGS = -std=gnu++11
LFLAGS = -lm

all: $(TARGET)

clean:
	rm $(DEPS) $(OBJECTS) $(TARGET)

.cpp.o:
	g++ -ggdb -MMD -MP $(CFLAGS) -c $< -o $@

$(TARGET): $(OBJECTS)
	g++ $(OBJECTS) -o $(TARGET) $(LFLAGS)

-include $(DEPS)
